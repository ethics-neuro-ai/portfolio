â–¶ï¸ Example Usage

from transformers import AutoTokenizer, AutoModelForSeq2SeqLM

tokenizer = AutoTokenizer.from_pretrained("models/flan-t5-small-finetuned")
model = AutoModelForSeq2SeqLM.from_pretrained("models/flan-t5-small-finetuned")

prompt = "Write a formal email requesting project update."
inputs = tokenizer(prompt, return_tensors="pt")
outputs = model.generate(**inputs, max_length=200)
email = tokenizer.decode(outputs[0], skip_special_tokens=True)
print(email)
